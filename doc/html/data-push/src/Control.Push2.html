-- | `Right` when the predicate returns `True`
pushP2 ::
     Push2 f Either => (a -> Bool) -> f a -> Pushed2 f Either (Either (f a) (f a))
pushP2 p =
  pmap2 $ \x ->
    if p x
      then Right x
      else Left x

-- | `fmap` and `push2`
pmap2 :: Push2 f g => (a -> g b c) -> f a -> Pushed2 f g (g (f b) (f c))
pmap2 f = push2 . fmap f
